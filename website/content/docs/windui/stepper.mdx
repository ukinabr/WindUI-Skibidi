## Stepper Element

Stepper is a numeric control with minus/plus buttons and optional dragging on the value track.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Amount = Tab:Stepper({
    Title = "Amount",
    Min = 0,
    Max = 10,
    Default = 4,
    Increment = 1,
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Volume = Tab:Stepper({
    Title = "Volume",
    Desc = "Drag or tap buttons",
    Value = {
        Min = 0,
        Max = 100,
        Default = 45,
        Increment = 5,
    },
    Width = 190,
    Buttons = true,
    Draggable = true,
    Format = function(value)
        return value .. "%"
    end,
    Callback = function(value)
        print("volume", value)
    end,
})
```
    </TabsContent>
</Tabs>
