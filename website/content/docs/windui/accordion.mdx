## Accordion Element

Accordion creates collapsible rows. Use `Multiple = true` to allow more than one open row.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Notes = Tab:Accordion({
    Title = "Notes",
    Items = {
        { Title = "Switch", Desc = "Page switch animations are enabled." },
        { Title = "Press", Desc = "Rows use press feedback." },
    },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local FAQ = Tab:Accordion({
    Title = "FAQ",
    Desc = "Collapsible help",
    Multiple = true,
    DefaultOpen = 1,
    Items = {
        { Title = "Mobile", Desc = "Touch friendly controls.", Icon = "smartphone" },
        { Title = "Config", Desc = "Use flags for saving.", Icon = "save" },
    },
})

FAQ:Open(2)
```
    </TabsContent>
</Tabs>
