## TabBox Element

TabBox creates small nested pages inside the current tab or section. Each page supports normal WindUI elements.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">With Elements</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Box = Tab:TabBox({
    Title = "Modes",
    Desc = "Nested pages",
})

local Main = Box:Tab({ Title = "Main", Icon = "home", Value = "main" })
local Logs = Box:Tab({ Title = "Logs", Icon = "history", Value = "logs" })

Main:Paragraph({ Title = "Ready", Desc = "Main page selected." })
Logs:Paragraph({ Title = "Events", Desc = "No logs yet." })
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Box = Section:TabBox({
    Title = "Build Panel",
    TabHeight = 38,
    MinTabWidth = 76,
    MaxTabWidth = 160,
    Value = "controls",
})

Box:Tab({
    Title = "Overview",
    Icon = "sparkles",
    Value = "overview",
    Elements = function(Page)
        Page:StatusCard({ Title = "Renderer", Status = "Success", Value = "Ready" })
    end,
})

local Controls = Box:Tab({ Title = "Controls", Icon = "sliders-horizontal", Value = "controls" })
Controls:Stepper({ Title = "Amount", Min = 0, Max = 10, Default = 4 })
```
    </TabsContent>
</Tabs>
