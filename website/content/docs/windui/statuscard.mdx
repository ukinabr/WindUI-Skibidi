## StatusCard Element

StatusCard displays a label with a colored status dot.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Status = Tab:StatusCard({
    Title = "Renderer",
    Status = "Success",
    Value = "Ready",
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Status = Tab:StatusCard({
    Title = "Server",
    Desc = "Realtime connection",
    Status = "Info",
    Value = "Connecting",
    Width = 150,
    Callback = function(status, value)
        print(status, value)
    end,
})

Status:SetStatus("Success", "Online")
```
    </TabsContent>
</Tabs>
