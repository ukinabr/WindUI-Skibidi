## SegmentedControl Element

SegmentedControl is best for switching between 2 to 5 short modes.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Mode = Tab:SegmentedControl({
    Title = "Mode",
    Options = { "Auto", "Manual", "Off" },
    Default = "Auto",
    Callback = function(value)
        print("mode", value)
    end,
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Mode = Tab:SegmentedControl({
    Title = "Renderer",
    Desc = "Switch render mode",
    Width = 240,
    Options = {
        { Title = "Auto", Value = "auto" },
        { Title = "Manual", Value = "manual" },
        { Title = "Off", Value = "off", Disabled = true },
    },
    Value = "manual",
})
```
    </TabsContent>
</Tabs>
