## RadioGroup Element

RadioGroup is a vertical single-select control with touch friendly rows.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Quality = Tab:RadioGroup({
    Title = "Quality",
    Options = { "Low", "Medium", "High" },
    Default = "Medium",
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Quality = Tab:RadioGroup({
    Title = "Quality",
    Desc = "Choose one preset",
    Width = 220,
    AllowNone = false,
    Options = {
        { Title = "Low", Value = "low" },
        { Title = "Medium", Value = "medium" },
        { Title = "Ultra", Value = "ultra", Disabled = true },
    },
    Value = "medium",
    Callback = function(value, option)
        print(value, option.Title)
    end,
})
```
    </TabsContent>
</Tabs>
