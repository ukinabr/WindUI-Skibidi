## KeyValue Element

KeyValue displays a compact list of labels and values.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Details = Tab:KeyValue({
    Title = "Runtime",
    Items = {
        { Title = "Executor", Value = "Delta" },
        { Title = "Theme", Value = "Dark" },
    },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Details = Tab:KeyValue({
    Title = "Build",
    Desc = "Current runtime info",
    Items = {
        { Key = "Version", Value = "1.6.65", Icon = "tag" },
        { Key = "Branch", Value = "main", Icon = "git-branch" },
        { Key = "Mode", Value = "loadstring" },
    },
})

Details:SetItems({
    { Key = "Version", Value = "1.6.66" },
})
```
    </TabsContent>
</Tabs>
