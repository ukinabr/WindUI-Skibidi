## Badge Element

Badge renders a colored pill on the right side of an element row.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Badge = Tab:Badge({
    Title = "Access",
    Value = "Premium",
    Variant = "Warning",
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Badge = Tab:Badge({
    Title = "Build",
    Desc = "Current release channel",
    Value = "Stable",
    Variant = "Success",
    Icon = "badge-check",
    Width = 110,
    Callback = function(value)
        print(value)
    end,
})
```
    </TabsContent>
</Tabs>
