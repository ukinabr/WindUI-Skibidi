## ActionList Element

ActionList renders a vertical list of clickable rows with icons, descriptions and optional value pills.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
Tab:ActionList({
    Title = "Quick Actions",
    Actions = {
        "Refresh",
        "Copy Link",
        "Reset",
    },
    Callback = function(action, index)
        print(index, action.Title)
    end,
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Actions = Tab:ActionList({
    Title = "Build Actions",
    Desc = "Small rows for common commands.",
    Actions = {
        { Title = "Refresh", Desc = "Reload data", Icon = "refresh-cw", Value = "R" },
        { Title = "Export", Desc = "Save current config", Icon = "download", Color = "#38bdf8" },
        { Title = "Locked", Icon = "lock", Disabled = true },
    },
    Callback = function(action, index)
        print("action", index, action.Title)
    end,
})
```
    </TabsContent>
</Tabs>
