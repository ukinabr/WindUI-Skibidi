## Timeline Element

Timeline renders ordered status events with rails and reveal animations.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
Tab:Timeline({
    Title = "Events",
    Items = {
        { Title = "Window opened", Value = "Success" },
        { Title = "Config loaded", Value = "Info" },
        { Title = "Ready", Value = "Success" },
    },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Events = Tab:Timeline({
    Title = "Deploy Flow",
    Desc = "Recent pipeline steps",
    Items = {
        { Title = "Build queued", Desc = "Waiting for worker", Value = "Info" },
        { Title = "Compiled", Desc = "Static pages generated", Value = "Success" },
        { Title = "Manual review", Desc = "Optional check", Color = "#f59e0b" },
    },
})

Events:SetItems({
    { Title = "Updated", Value = "Success" },
})
```
    </TabsContent>
</Tabs>
