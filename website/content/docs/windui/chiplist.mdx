## ChipList Element

ChipList displays compact pill rows. It supports multi-select or single-select mode.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Tags = Tab:ChipList({
    Title = "Tags",
    Options = { "Mobile", "Fast", "Glass" },
    Value = { "Mobile" },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Tags = Tab:ChipList({
    Title = "Theme Tags",
    Desc = "Compact tag selector",
    Width = 190,
    Multi = true,
    Options = {
        { Title = "Mobile", Value = "mobile" },
        { Title = "Motion", Value = "motion" },
        { Title = "Legacy", Value = "legacy", Disabled = true },
    },
    Value = { "mobile", "motion" },
    Callback = function(values)
        print(values)
    end,
})
```
    </TabsContent>
</Tabs>
