## CheckboxGroup Element

CheckboxGroup is a multi-select group for option sets.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Filters = Tab:CheckboxGroup({
    Title = "Filters",
    Options = { "Mobile", "PC", "Glass" },
    Value = { "Mobile" },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Filters = Tab:CheckboxGroup({
    Title = "Filters",
    Desc = "Select every matching tag",
    Width = 220,
    Options = {
        { Title = "Mobile", Value = "mobile" },
        { Title = "PC", Value = "pc" },
        { Title = "Legacy", Value = "legacy", Disabled = true },
    },
    SelectedValues = { "mobile" },
    Callback = function(values)
        print(table.concat(values, ", "))
    end,
})
```
    </TabsContent>
</Tabs>
