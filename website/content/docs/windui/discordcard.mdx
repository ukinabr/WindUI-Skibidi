## DiscordCard Element

DiscordCard is a styled invite block with copy and join actions. It normalizes invite codes into Discord URLs.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Card = Tab:DiscordCard({
    Title = "WindUI Community",
    Invite = "ftgs-development-hub-1300692552005189632",
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Card = Tab:DiscordCard({
    Title = "WindUI Community",
    Desc = "Copy the invite or run a custom join callback.",
    Invite = "https://discord.gg/ftgs-development-hub-1300692552005189632",
    Icon = "message-circle",
    Members = "10k+",
    Online = "Live",
    CopyTitle = "Copy Link",
    JoinTitle = "Join",
    Height = 156,
    Callback = function(url, card)
        setclipboard(url)
    end,
})
```
    </TabsContent>
</Tabs>
