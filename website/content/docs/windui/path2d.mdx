## Path2D Element

Path2D renders a normalized 0-1 coordinate path with animated segments, dots, labels and a moving marker.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Route = Tab:Path2D({
    Title = "Control Path",
    Points = {
        Vector2.new(0.08, 0.72),
        Vector2.new(0.32, 0.48),
        Vector2.new(0.68, 0.28),
        Vector2.new(0.92, 0.44),
    },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Route = Tab:Path2D({
    Title = "Delivery Route",
    Desc = "Animated route drawing with a moving marker.",
    Height = 170,
    Thickness = 5,
    Duration = 1.4,
    StepDelay = 0.04,
    Loop = true,
    AutoPlay = true,
    Labels = {
        { Point = 1, Text = "Start" },
        { Point = 3, Text = "Sync" },
        { Point = 4, Text = "Done" },
    },
    Points = {
        { 0.08, 0.72 },
        { 0.32, 0.48 },
        { 0.56, 0.62 },
        { 0.86, 0.34 },
    },
})
```
    </TabsContent>
</Tabs>
