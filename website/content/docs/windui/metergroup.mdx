## MeterGroup Element

MeterGroup shows multiple labeled progress meters in one element.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
local Meters = Tab:MeterGroup({
    Title = "Performance",
    Meters = {
        { Title = "CPU", Value = 42, Max = 100 },
        { Title = "Memory", Value = 1.4, Max = 2 },
    },
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Meters = Tab:MeterGroup({
    Title = "Renderer",
    Desc = "Live status values",
    Meters = {
        {
            Title = "Load",
            Desc = "Current render pressure",
            Value = 72,
            Max = 100,
            Color = "#0ea5e9",
        },
        {
            Title = "Cache",
            Value = 18,
            Max = 24,
            Format = function(value, max)
                return value .. "/" .. max
            end,
        },
    },
})

Meters:SetValue(1, 88)
```
    </TabsContent>
</Tabs>
