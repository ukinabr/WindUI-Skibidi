## EmptyState Element

EmptyState is a centered placeholder for empty lists, errors or no-results screens.

<Tabs defaultValue="basic">
    <TabsList>
        <TabsTrigger value="basic">Basic</TabsTrigger>
        <TabsTrigger value="full">Full</TabsTrigger>
    </TabsList>
    <TabsContent value="basic">
```luau
Tab:EmptyState({
    Title = "No results",
    Desc = "Try another filter.",
})
```
    </TabsContent>
    <TabsContent value="full">
```luau
local Empty = Tab:EmptyState({
    Title = "No configs",
    Desc = "Save a config to see it here.",
    Icon = "folder-open",
    IconSize = 36,
    Height = 160,
    Buttons = {
        {
            Title = "Create Config",
            Icon = "plus",
            Callback = function()
                print("create")
            end,
        },
    },
})
```
    </TabsContent>
</Tabs>
